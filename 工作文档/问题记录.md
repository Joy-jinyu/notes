function openDialog(dialogClassName, callback) {
    dialog[dialogClassName] = layer.open({
        type: 1,
        closeBtn: 0,
        content: $(`.${dialogClassName}`),
        area: ['auto', 'auto'],
        shadeClose: false,
        title: false,
        scrollbar: false,
        success($root) {
            $root.one('click', `.${dialogClassName} .close, .${dialogClassName} .close-btn`, () => {
                layer.close(dialog[dialogClassName]);
            })
            callback && callback($root)
        }
    })
}
